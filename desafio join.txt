
CREATE TABLE Jogadores (
    id_jogador INT PRIMARY KEY,
    nome VARCHAR(50)
);

CREATE TABLE Jogos (
    id_jogo INT PRIMARY KEY,
    titulo VARCHAR(50)
);

CREATE TABLE Compras (
    id_compra INT PRIMARY KEY,
    id_jogador INT,
    id_jogo INT
);

SELECT Jogadores.nome, Jogos.titulo
FROM Compras
INNER JOIN Jogadores ON Compras.id_jogador = Jogadores.id_jogador
INNER JOIN Jogos ON Compras.id_jogo = Jogos.id_jogo;


SELECT Jogadores.nome, Jogos.titulo
FROM Jogadores
LEFT JOIN Compras ON Jogadores.id_jogador = Compras.id_jogador
LEFT JOIN Jogos ON Compras.id_jogo = Jogos.id_jogo;


SELECT Jogos.titulo, Jogadores.nome
